Advanced Options

<!-- Advanced Snackbar Options -->
<script>
// Persistent snackbar (no auto-dismiss)
snackbar.show('Persistent message', {
    duration: 0,
    dismissible: true
});

// With action button
snackbar.show('File deleted', {
    action: {
        text: 'Undo',
        handler: 'undoDelete()'
    }
});

// Custom duration (10 seconds)
snackbar.show('Long message', {
    duration: 10000
});

// Custom icon and styling
snackbar.show('Custom notification', {
    icon: 'üéâ',
    variant: 'success',
    position: 'top-center'
});

// Dismiss all snackbars
snackbar.dismissAll();
</script>
